58  template<class T>
59  void solve(T &A, Vector &x, Vector &b)
60  {
61  // Variables
62  _timerP.tic(); // Comenzamos a medir el tiempo
63  int j = 0;
64  double diag = 0.0;
65  double tol2 = _tolP*_tolP;
66  int n = A.gN();
67  Vector x0(x.size());
68  x0 = x;
69  // Comenzamos con el metodo de Jacobi
70  for (int k = 0; k < _iterMaxP; ++k)
71  {
72  for (int i = 0; i < n; ++i)
73  {
74  x[i] = 0.0;
75  //std::cout << i << ": " << x.asizeVect << std::endl;
76  for (int l = A.gIrow(i); l < A.gIrow(i+1); ++l)
77  {
78  j = A.gCol(l);
79  if( i != j)
80  {
81  x[i] += A.gData(l)*x0[j];
82  }
83  else
84  {
85  diag = A.gData(l);
86  }
87  }
88  _errP += (b[i]-x[i]-diag*x0[i])*(b[i]-x[i]-diag*x0[i]);
89  x[i] = (b[i]-x[i])/diag;
90  }
91  //std :: cout << k+1 << ": " << _errP << " " << tol2 << std :: endl;
92  // Iteraciones realizadas
93  _iterP++;
94  // Revisamos la tolerancia de paro
95  if (_errP < tol2) break;
96  x0 = x;
97  }
98  //x.print();
99  // Terminamos de medir el tiempo usado
100  _errP = sqrt(_errP);
101  _timerP.toc();
102  _etimeP = _timerP.etime();
103  }
